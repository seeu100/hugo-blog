pacman -S docker #安装

docker pull archlinux #拉取镜像
docker pull mogeko/archlinux
docker build -f 1now/archlinux-ssh.dockfile -t archlinuxssh:v1 #从dockfile构建

docker cp  hostfile_path 0711eef99e72:guestfile_path

docker images #查看镜像
docker ps #查看容器
docker ps -a #查看所有容器
docker info #查看docker信息
docker network ls #网络

docker run -d -p 2222:22 -p 5921:5901 -v /home/erwa/share:/home/erwa/share  --name arch -v /tmp/.X11-unix:/tmp/.X11-unix -e DISPLAY=unix$DISPLAY -e GDK_SCALE -e GDK_DPI_SCALE  erwa/archlinuxssh:1.1 /usr/bin/sshd -D # 启动

docker commit -m "update" -a "erwa" arch erwa/archlinuxssh:1.3 #更新镜像

docker exec -it arch bash # 进入bash
docker exec -it --user root ee880db9ae68  /bin/bash
docker exec -it --user erwa ee880db9ae68  /bin/bash

vncviewer 127.0.0.1:5921 # 打开VNC界面（安装tigervnc）
ssh erwa@127.0.0.1 -p 2222 # SSH

docker stop arch && docker rm arch # 停止并删除
docker start arch  #容器停止后启动
docker pause clever_dubinsky #暂停
docker rm erwa/archlinux #删除容器
docker rmi 0e7f69359602 #删除镜像

ssh erwa@127.0.0.1 -p 2222
sudo emerge -av remmina 

sudo rm /var/lib/docker #删除docker所有文件


docker save -o archlinuxssh.tar erwa/archlinuxssh:1.3 #导出
docker load -i archlinuxssh.tar #导入
