# gem install optparse
require 'ostruct'
require 'optparse'

options = OpenStruct.new
OptionParser.new do |opts|
  opts.on('-n', '--name name', 'name') {
     |v| options[:name] = v }
  opts.on('-a', '--age age', 'age') {
     |v| options[:age] = v }
end.parse!

puts "name is #{
      options[:name]}, age is #{
      options[:age]}"
